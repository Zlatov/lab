Install

  • ##sudo apt-get install bootstrap-sass
  • sudo apt-get install sass
  • sudo npm install -g sass установит в /usr/local/bin/sass

simply start

file: test.scss

$font-stack:    Helvetica, sans-serif;
$primary-color: #333;

body {
  font: 100% $font-stack;
  color: $primary-color;
}

run:

sass test.scss test.css

file example2.scss:

@import './test.scss';

or: (folder with *.scss)

@import './';

run:

sass example2.scss test.css
