Медиа запросы

min-width: 768px === Правило применяется к браузеру, ширина которого больше значения, указанного в запросе.

max-width: 1024px === Правило применяется к браузеру, ширина которого меньше значения, указанного в запросе.

/*@media (min-width: 768px) and (max-width: 991px) {*/
@media (min-width: 768px) {
    .navbar-nav li:first-child a {
        padding-left: 0;
    }
}

Флекс боксы

Стили для контейнера

.flex {
	display: flex;

	/* Ось */
	/* расположить боксы горизонтально */
	flex-direction: row;
	/* вертикально */
	flex-direction: column;

	/* Выравнивание вдоль оси (подобно text-align если боксы горизонтально) */
	justify-content: flex-start;
	justify-content: flex-end;
	justify-content: center;
	justify-content: space-between;
	justify-content: space-around;

	/* Выравнивание поперек оси */
	align-items: stretch;
	align-items: flex-start;
	align-items: flex-end;
	align-items: center;
	align-items: baseline;

	/* Многострочность */
	flex-wrap: nowrap;
	flex-wrap: wrap;

	/* Выравнивани многострочности вдоль оси многострочности (поперек главной оси) */
	align-content: stretch;
	align-content: flex-start;
	align-content: flex-end;
	align-content: center;
	align-content: space-between;
	align-content: space-around;
}
