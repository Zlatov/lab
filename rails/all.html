  def self.get_hash id
    begin
      query = ActiveRecord::Base.sanitize_sql_array(["CALL get_seo(:id);", id: id])
      result = connection.execute(query)
    ensure
      connection.close
    end
    results = []
    result.each(as: :hash) {|row| results << row}
    results.first || {}
  end

  def self.set_hash hash
    hash = {title:nil,description:nil,keywords:nil,h1:nil}.merge! hash
    query = ActiveRecord::Base.sanitize_sql_array(["CALL set_seo(:id, :title, :description, :keywords, :h1);", hash])
    connection.execute(query)
  end

end

Экспорт/импорт схемы

Настройка экспорта схемы в файле config/environments/development.rb:

  # Если выбрано :ruby, тогда схема хранится в db/schema.rb, иначе в db/structure.sql
  config.active_record.schema_format = :sql

rails db:structure:dump - экспорт схемы.

rails db:structure:load - импорт схемы.
