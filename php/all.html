  • intval

  • htmlspecialchars

Интеджер

if ($number)
{

Стринг

Не предполагаем вхождение html тегов:

$input_text = strip_tags($_GET['input_text']);
$input_text = htmlspecialchars($input_text);
$input_text = mysql_escape_string($input_text);

$checkbox_arr = array_map('intval', $_POST['checkbox']);
